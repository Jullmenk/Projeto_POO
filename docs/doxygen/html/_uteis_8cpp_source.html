338void Uteis::DevolverLivro(Biblioteca& bib){
339 string id_Pessoa,id_Livro;
340 bool findUser=false,findLivro=false;
341 Pessoa *pessoa=nullptr;
342 Geral *livro=nullptr;
343 retorno user = RetornarType_String_User();
344 while (!findUser)
345 {
346 if(!bib.listarLeitores(true,user.categoria))return;
347 cout << "\t Digite o NIF do utilizador : ";
348 cin>> id_Pessoa;
349 pessoa = bib.ProcurarUtilizador(id_Pessoa);
350 if(pessoa)findUser=true;
351 }
352
353 pessoa->listarEmprestimos();
354
355 if(pessoa->getNumeroDeEmprestimosTotais()<=0)return;
356
357 while (!findLivro)
358 {
359 cout<< "\tDigite o ISBN/ISSN do Livro Deseja Devolver : ";
360 cin >> id_Livro;
361 livro = bib.ProcurarLivro(id_Livro);
362 if(livro)findLivro=true;
363
364 }
365
366 auto& empresLista = bib.getEmprestimosPorCategoria();
367 auto it = empresLista.find(livro->getCategoria());
368 if (it == empresLista.end()) {
369 cout << "Nenhum livro dessa categoria foi requisitado.\n";
370 return;
371 }
372
373 // Iteração de trás para frente com iteradores reversos
374 for (auto rit = it->second.rbegin(); rit != it->second.rend(); ++rit) {
375 auto& emprestimo = *rit;
376 if ((emprestimo.getNifEmprestimo() == pessoa->getNIF()) && (emprestimo.getIDLivroEmprestimo() == livro->getCodigo())) {
377 livro->setDisponivel();
378 pessoa->decrementarEmprestimosAtivos();
379 time_t dataAtual = time(nullptr);
380 if(emprestimo.getDataDevolucaoEmprestimo() < dataAtual){
381 pessoa->incrementarMulta();
382 pessoa->EnviarNotificacoesdeAtraso();
383 system("pause");
384 }
385 cout << "Livro devolvido com sucesso.\n";
386 bib.transformarReservaEmEmprestimo(emprestimo.getCategoriaLivroEmprestimo(),emprestimo.getIDLivroEmprestimo());
387 system("pause");
388 return;
389 }
390 }
